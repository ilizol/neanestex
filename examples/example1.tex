% Prints a modekey with the greek "echos"
\newenvironment{modekey}{%
    \begin{center}%
        \color[HTML]{800000}%
        \fontsize{20bp}{\baselineskip}%
        \fontspec{Neanes.otf}
        \byzglyph{modeWordEchos}%
}%
{%
    \end{center}%
}

% Prints a modekey with the English "Mode"
\newenvironment{modekeyenglish}{%
    \begin{center}%
        \color[HTML]{800000}%
        \fontsize{18bp}{\baselineskip}\selectfont%
        \raisebox{7bp}{Mode}%
        \fontsize{20bp}{\baselineskip}\selectfont%
        \fontspec{Neanes}%
}%
{%
    \end{center}%
}
